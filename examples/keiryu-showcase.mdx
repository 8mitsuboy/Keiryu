#### Component

<div tone="quiet">
  MDX keeps markdown flow while allowing JSX islands.
</div>

```tsx
export function ThemeNote(props: { tone: 'quiet' | 'sharp'; children: React.ReactNode }) {
  return <aside data-tone={props.tone}>{props.children}</aside>
}
```
